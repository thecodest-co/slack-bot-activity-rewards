1   package com.thecodest.slack.activityrewards.slack;

3   import com.slack.api.methods.response.chat.ChatPostMessageResponse;
4   import com.thecodest.slack.activityrewards.MessageBeing;
5   import java.time.LocalDateTime;
6   import java.util.List;
7   
8   /**
9    * Simplified Slack interface. If you want to use some additional APIs then you
10   * need to add that method here.
11   * Idea was that to have a simple „only things that we need” interface.
12   */
13  public interface SlackClient {
14  
15  	/**
16  	 * Send message to user. Message text is send as it is. If you need format your
17  	 * message you need to send proper
18  	 * formatting tags here.
19  	 *
20  	 * @param slackUser
21  	 *            target user.
22  	 * @param message
23  	 *            message as text.
24  	 * @return server response.
25  	 */
26  	ChatPostMessageResponse sendMessageToUser(SlackUser slackUser, String message);
27  
28  	/**
29  	 * Send message to channel (or user). Message text is send as it is. If you need
30  	 * format your
31  	 * message you need to send proper
32  	 * formatting tags here.
33  	 *
34  	 * @param slackChannel
35  	 *            target channel.
36  	 * @param message
37  	 *            message as text.
38  	 * @return server response.
39  	 */
40  	ChatPostMessageResponse sendMessageToChannel(String slackChannel, String message);
41  
42  	/**
43  	 * Find all users that emails match to given list.
44  	 *
45  	 * @param emails
46  	 *            list of user emails that se looking for.
47  	 * @return List of @{@link SlackUser}s that we found.
48  	 */
49  	List<SlackUser> getUsersByEmails(List<String> emails);
50  
51  	/**
52  	 * Find all users that IDs match to given list.
53  	 *
54  	 * @param ids
55  	 *            list of user ids that se looking for.
56  	 * @return List of @{@link SlackUser}s that we found.
57  	 */
58  	List<SlackUser> getUsersByIds(List<String> ids);
59  
60  	/**
61  	 * Return all messages from channel with given ID
62  	 *
63  	 * @param channelId
64  	 *            id of slack channel
65  	 * @param from
66  	 * @param to
67  	 * @return
68  	 */
69  	List<MessageBeing> getMessagesFromChannel(String channelId, LocalDateTime from, LocalDateTime to);
70  
71  }
