1   package com.thecodest.slack.activityrewards.aws;
2   
3   import com.amazonaws.services.s3.AmazonS3;
4   import com.amazonaws.services.s3.model.PutObjectRequest;
5   import com.amazonaws.services.s3.model.PutObjectResult;
6   import com.fasterxml.jackson.databind.ObjectMapper;
7   import com.thecodest.slack.activityrewards.MonthlyResults;
8   import io.vavr.control.Try;
9   import java.nio.file.Files;
10  import java.nio.file.Path;
11  import java.nio.file.Paths;
12  import javax.inject.Inject;
13  import lombok.extern.java.Log;
14  
15  @Log
16  class CustomS3PersistenceImpl implements CustomS3Persistence {
17  
18  	private static final String BUCKET_NAME = "activity-reward";
19  	private final AmazonS3 amazonS3;
20  	private final ObjectMapper objectMapper;
21  
22  	@Inject
23  	CustomS3PersistenceImpl(AmazonS3 amazonS3, ObjectMapper objectMapper) {
24  		this.amazonS3 = amazonS3;
25  		this.objectMapper = objectMapper;
26  	}
27  
28  	@Override
29  	public void save(MonthlyResults results) {
30  		if (!amazonS3.doesBucketExistV2(BUCKET_NAME)) {
31  			amazonS3.createBucket(BUCKET_NAME);
32  		}
33  		var awsOperationResult = Try
34  				.of(() -> Paths.get("./", "target", results.year() + "-" + results.month().toString() + ".json"))
35  				.mapTry(path -> {
36  					if (!path.toFile().exists())
37  						return Files.createFile(path);
38  					return path;
39  				})
40  				.map(Path::toFile)
41  				.andThenTry(f -> objectMapper.writeValue(f, results))
42  				.map(file -> new PutObjectRequest(BUCKET_NAME, results.year() + "/" + results.month(), file))
43  				.map(putObjectRequest -> amazonS3.putObject(putObjectRequest))
44  				.get();
45  		log.info(awsOperationResult.getVersionId());
46  	}
47  }
