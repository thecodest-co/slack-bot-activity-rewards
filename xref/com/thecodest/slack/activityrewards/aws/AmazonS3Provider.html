1   package com.thecodest.slack.activityrewards.aws;
2   
3   import com.amazonaws.auth.AWSCredentials;
4   import com.amazonaws.auth.AWSStaticCredentialsProvider;
5   import com.amazonaws.regions.Regions;
6   import com.amazonaws.services.s3.AmazonS3;
7   import com.amazonaws.services.s3.AmazonS3ClientBuilder;
8   import javax.inject.Inject;
9   import javax.inject.Named;
10  import javax.inject.Provider;
11  
12  class AmazonS3Provider implements Provider<AmazonS3> {
13  
14  	private final AWSCredentials credentials;
15  	private final String region;
16  
17  	@Inject
18  	AmazonS3Provider(AWSCredentials credentials, @Named("aws-region") String region) {
19  		this.credentials = credentials;
20  		this.region = region;
21  	}
22  
23  	@Override
24  	public AmazonS3 get() {
25  		return AmazonS3ClientBuilder.standard()
26  				.withCredentials(new AWSStaticCredentialsProvider(credentials))
27  				.withRegion(Regions.fromName(region))
28  				.build();
29  	}
30  }
